SUCCESS (200): { thresholds: { positiveThreshold: number, warningThreshold: number, criticalThreshold: number, targetRoi: number|null, alertOnNegative: boolean, alertOnTargetMiss: boolean }, updatedAt: timestamp }
ERRORS:
- 400: invalid field values, threshold ordering violated, empty body
- 401: unauthenticated
